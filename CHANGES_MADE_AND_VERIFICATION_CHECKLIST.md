```sql
-- Check if after-hours changes are in historical data
SELECT 
    h.TradingDate,
    h.Strike,
    h.OptionType,

    h.UpperCircuitLimit AS Historical_UC,
    m.MaxGlobalSeq,
    m.LastRecordTime
```
